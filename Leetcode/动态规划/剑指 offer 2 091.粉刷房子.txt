class Solution {
public:
    int minCost(vector<vector<int>>& costs) {
        int dp[105][3]={0};
        dp[0][0]=costs[0][0];
        dp[0][1]=costs[0][1];
        dp[0][2]=costs[0][2];
        for(int i=1;i<costs.size();i++){
            for(int j=0;j<3;j++){
                 dp[i][j]=min(dp[i-1][(j+1)%3],dp[i-1][(j+2)%3])+costs[i][j];
            }
        }     
        return min(min(dp[costs.size()-1][0],dp[costs.size()-1][1]),dp[costs.size()-1][2]);
    }
};
